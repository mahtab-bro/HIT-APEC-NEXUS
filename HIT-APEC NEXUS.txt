// ═══════════════════════════════════════════════════════════════
async function sha256(str){
  const buf = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(str));
  return Array.from(new Uint8Array(buf)).map(b=>b.toString(16).padStart(2,'0')).join('');
}

// ═══════════════════════════════════════════════════════════════
// STATE ENGINE
// ═══════════════════════════════════════════════════════════════
const S = (()=>{
  let STORE_KEY = 'nexus3_state'; // default; overridden per-user on login/signup
  let _s = null;
  let _listeners = [];

  function fresh(){
    return {
      user: { name:'', email:'', passHash:'', joinedAt: Date.now() },
      xp: { scholar:0, fitness:0, community:0, mind:0, skill:0 },
      gold: 0,
      level: 0,
      streak: { count:0, lastProofDate:'' },
      proofs: [],         // [{id, questId, cat, xp, gold, imgUrl, ts, note}]
      cooldowns: {},      // {questId: timestamp_until}
      badges: []
    };
  }

  function load(){
    try{ const d=localStorage.getItem(STORE_KEY); if(d) return JSON.parse(d); }catch(e){}
    return fresh();
  }
  function save(){ localStorage.setItem(STORE_KEY, JSON.stringify(_s)); }

  function notify(){ _listeners.forEach(fn=>fn(_s)); }

  // Track the current session's store key so logout knows what key is active
  let _activeKey = null;
  function getActiveKey(){ return _activeKey; }

  return {
    init(storeKey){
      if(storeKey) STORE_KEY = storeKey;
      _activeKey = STORE_KEY;
      _s = load(); return _s;
    },
    get(){ return _s; },
    on(fn){ _listeners.push(fn); },
    update(fn){ fn(_s); save(); notify(); },
    reset(){ _s = fresh(); save(); notify(); },
    getActiveKey
  };
})();

// ACCOUNT REGISTRY — persists all signed-up accounts on this device
// Key: nexus3_accounts  →  { [email]: { passHash, stateKey } }
// Each user's full state lives under their own stateKey.
// ═══════════════════════════════════════════════════════════════
const ACCOUNTS = (()=>{
  const KEY = 'nexus3_accounts';
  function load(){ try{ const d=localStorage.getItem(KEY); if(d) return JSON.parse(d); }catch(e){} return {}; }
  function save(map){ localStorage.setItem(KEY, JSON.stringify(map)); }
  return {
    exists(email){ return !!load()[email.toLowerCase()]; },
    register(email, passHash){
      const map=load();
      const key='nexus3_state_'+email.toLowerCase().replace(/[^a-z0-9]/g,'_');
      map[email.toLowerCase()]={passHash, stateKey:key};
      save(map);
      return key;
    },
    verify(email, passHash){
      const map=load();
      const rec=map[email.toLowerCase()];
      if(!rec) return null;          // not found
      if(rec.passHash!==passHash) return false; // wrong password
      return rec.stateKey;           // success → return stateKey
    }
  };
})();

// ═══════════════════════════════════════════════════════════════
// LEADERBOARD (shared across sessions in same browser via a
// separate storage key — honest about being single-device for now)
// ═══════════════════════════════════════════════════════════════
const LB = (()=>{
  const KEY = 'nexus3_lb';
  function load(){ try{ const d=localStorage.getItem(KEY); if(d) return JSON.parse(d); }catch(e){} return []; }
  function save(entries){ localStorage.setItem(KEY, JSON.stringify(entries)); }

  return {
    upsert(userId, entry){
      const all = load();
      const idx = all.findIndex(e=>e.id===userId);
      if(idx>=0) all[idx]={...all[idx],...entry};
      else all.push({id:userId,...entry});
      save(all);
    },
    get(){ return load().sort((a,b)=>b.xp-a.xp); }
  };
})();

const Toast = (()=>{
  const cols = {success:'#B8F53A',error:'#EF4444',warn:'#F59E0B',info:'#60A5FA'};
  const ics  = {success:'fa-check-circle',error:'fa-times-circle',warn:'fa-exclamation-triangle',info:'fa-info-circle'};
  function show({title,body='',type='info',dur=4000}){
    const z = document.getElementById('toast-zone');
    const t = document.createElement('div');
    t.className='toast';
    t.style.borderLeft=`3px solid ${cols[type]||cols.info}`;
    t.innerHTML=`<i class="fa ${ics[type]||ics.info}" style="color:${cols[type]};font-size:16px;margin-top:1px;flex-shrink:0;"></i><div style="flex:1;"><div style="font-weight:700;margin-bottom:2px;">${title}</div>${body?`<div style="color:var(--muted);font-size:12px;">${body}</div>`:''}</div><button onclick="this.parentElement.remove()" style="background:none;border:none;color:var(--muted);cursor:pointer;padding:0 2px;font-size:15px;">×</button>`;
    z.appendChild(t);
    setTimeout(()=>{ t.classList.add('out'); setTimeout(()=>t.remove(),250); }, dur);
  }
  return {show};
})();

function totalXP(s){ return Object.values(s.xp).reduce((a,b)=>a+b,0); }

function getLevel(xp){ return Math.floor(xp / CFG.XP_PER_LEVEL); }

function getRankLabel(level){
  if(level===0) return 'Newcomer';
  if(level<=2)  return 'Initiate';
  if(level<=5)  return 'Operative';
  if(level<=9)  return 'Specialist';
  if(level<=14) return 'Commander';
  return 'Legend';
}

function todayStr(){ return new Date().toISOString().slice(0,10); }

function timeAgo(ts){
  const d=Date.now()-ts;
  if(d<60000) return 'just now';
  if(d<3600000) return Math.floor(d/60000)+'m ago';
  if(d<86400000) return Math.floor(d/3600000)+'h ago';
  return Math.floor(d/86400000)+'d ago';
}

function isOnCooldown(questId){
  const cd = S.get().cooldowns[questId];
  return cd && cd > Date.now();
}

function cdRemaining(questId){
  const rem = (S.get().cooldowns[questId]||0) - Date.now();
  if(rem<=0) return '';
  const h=Math.floor(rem/3600000), m=Math.floor((rem%3600000)/60000);
  return `${h}h ${m}m`;
}

function floatXP(amount){
  const el = document.createElement('div');
  el.style.cssText=`position:fixed;bottom:80px;right:30px;font-family:'Space Mono',monospace;font-size:22px;font-weight:700;color:var(--lime);pointer-events:none;z-index:9999;animation:xpFloat 1.2s ease-out forwards;`;
  el.textContent=`+${amount} XP`;
  document.body.appendChild(el);
  setTimeout(()=>el.remove(),1200);
}

function animCount(el, from, to, ms=700){
  const start=performance.now();
  function step(now){
    const p=Math.min((now-start)/ms,1);
    const ease=1-Math.pow(1-p,3);
    el.textContent=Math.round(from+(to-from)*ease).toLocaleString();
    if(p<1) requestAnimationFrame(step);
  }
  requestAnimationFrame(step);
}

async function uploadToImgBB(file){
  if(!CFG.IMGBB_KEY){
    // No key — return a placeholder URL for demo purposes
    return URL.createObjectURL(file); // local blob URL (works for demo session)
  }
  const form = new FormData();
  form.append('image', file);
  const res = await fetch(`https://api.imgbb.com/1/upload?key=${CFG.IMGBB_KEY}`, {
    method:'POST', body:form
  });
  const data = await res.json();
  if(!data.success) throw new Error(data.error?.message||'Upload failed');
  return data.data.url;
}

async function handleAuth(){
  const email=document.getElementById('inp-email').value.trim();
  const pass=document.getElementById('inp-pass').value;
  const nameVal=document.getElementById('inp-name').value.trim();
  const errEl=document.getElementById('email-err');

  if(!email||!pass){ Toast.show({title:'Fill in all fields.',type:'error'}); return; }
  if(!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)){ errEl.style.display='block'; errEl.textContent='Enter a valid email.'; return; }
  if(pass.length<6){ Toast.show({title:'Password must be at least 6 characters.',type:'error'}); return; }
  errEl.style.display='none';

  const btn=document.getElementById('auth-btn');
  btn.innerHTML='<div class="spinner"></div>';
  btn.disabled=true;

  const passHash = await sha256(pass+email); // salted with email

  if(authTab==='signup'){
    // Block duplicate signups
    if(ACCOUNTS.exists(email)){
      btn.innerHTML='<span id="auth-btn-text">Create Account</span>';
      btn.disabled=false;
      Toast.show({title:'This email is already registered.',body:'Please sign in instead.',type:'error'});
      return;
    }
    const stateKey = ACCOUNTS.register(email, passHash);
    S.init(stateKey);
    const name = nameVal || email.split('@')[0].replace(/[._]/g,' ').replace(/\b\w/g,c=>c.toUpperCase());
    S.update(st=>{
      st.user.name=name;
      st.user.email=email;
      st.user.passHash=passHash;
      st.user.joinedAt=Date.now();
    });
    bootApp();
  } else {
    // Login: look up account in registry
    const result = ACCOUNTS.verify(email, passHash);
    if(result===null){
      btn.innerHTML='<span id="auth-btn-text">Sign In</span>';
      btn.disabled=false;
      Toast.show({title:'No account found for this email.',body:'Please create an account first.',type:'error'});
      return;
    }
    if(result===false){
      btn.innerHTML='<span id="auth-btn-text">Sign In</span>';
      btn.disabled=false;
      Toast.show({title:'Incorrect password.',type:'error'});
      return;
    }
    // result is the stateKey — load that user's state
    S.init(result);
    bootApp();
  }
}

function bootApp(){
  const s=S.get();
  // Sync leaderboard
  LB.upsert(s.user.email, {
    name:s.user.name,
    xp:totalXP(s),
    level:getLevel(totalXP(s)),
    proofCount:s.proofs.length
  });

  const auth=document.getElementById('auth-screen');
  auth.style.transition='opacity .4s';
  auth.style.opacity='0';
  setTimeout(()=>{
    auth.style.display='none';
    document.getElementById('app').style.display='block';
    initApp();
  },400);
}

function logout(){
  // Clear only the active session marker — user data stays in storage
  document.getElementById('app').style.display='none';
  const auth=document.getElementById('auth-screen');
  auth.style.display='flex';
  auth.style.opacity='1';
  // Reset auth form
  setAuthTab('login');
  document.getElementById('auth-btn').innerHTML='<span id="auth-btn-text">Sign In</span>';
  document.getElementById('auth-btn').disabled=false;
  Toast.show({title:'Logged out.',body:'Sign in to return to your account.',type:'info'});
}

const PAGE_BG_FILES = {
  dashboard:   'Harbin_Institute_of_Technology_(Shenzhen)_Old_Building_(Bachelor).jpg',
  quests:      'Harbin_IT_04.jpg',
  leaderboard: 'Main_Building_HIT.jpg',

const _bgCache = {};

async function resolveWikimediaUrl(filename){
  const api = `https://commons.wikimedia.org/w/api.php?action=query&titles=File:${encodeURIComponent(filename)}&prop=imageinfo&iiprop=url&iiurlwidth=1280&format=json&origin=*`;
  const res = await fetch(api);
  const data = await res.json();
  const pages = data.query.pages;
  const page = pages[Object.keys(pages)[0]];
  return (page.imageinfo[0].thumburl || page.imageinfo[0].url);
}

function applyBackground(url){
  document.getElementById('main').style.backgroundImage = url ? `url('${url}')` : 'none';
}

async function preloadBackgrounds(){
  for(const [pageName, filename] of Object.entries(PAGE_BG_FILES)){
    resolveWikimediaUrl(filename).then(url=>{
      _bgCache[pageName] = url;
      if(currentPage === pageName) applyBackground(url);
      const img = new Image(); img.src = url;
    }).catch(()=>{});
  }
}

function setPageBackground(page){
  if(_bgCache[page]) applyBackground(_bgCache[page]);
  else applyBackground('');
}

function navigate(page){
  currentPage=page;
  setPageBackground(page);
  document.querySelectorAll('.page').forEach(p=>p.style.display='none');
  document.querySelectorAll('.nav-item').forEach(n=>n.classList.remove('active'));
  document.getElementById('page-'+page).style.display='block';
  const nav=document.getElementById('nav-'+page);
  if(nav) nav.classList.add('active');
  if(page==='dashboard') renderDashboard();
  if(page==='quests') renderQuests();
  if(page==='leaderboard') renderLeaderboard();
  if(page==='profile') renderProfile();
}

function previewProof(event, questId){
  const file=event.target.files[0];
  if(!file) return;
  if(file.size>33554432){ Toast.show({title:'File too large. Max 32MB.',type:'error'}); return; }
  _pendingFile=file;
  _pendingPreviewUrl=URL.createObjectURL(file);
  document.getElementById('upload-placeholder').style.display='none';
  document.getElementById('upload-preview').style.display='block';
  document.getElementById('proof-img-preview').src=_pendingPreviewUrl;

async function submitProof(questId){
  if(!_pendingFile){ Toast.show({title:'Upload a proof image first.',type:'warn'}); return; }
  const q=QUEST_DEFS.find(x=>x.id===questId);
  if(!q) return;

  const btn=document.getElementById('submit-proof-btn');
  const errEl=document.getElementById('submit-error');
  btn.innerHTML='<div class="spinner"></div> Uploading...';
  btn.disabled=true;
  errEl.style.display='none';

  try{
    const imgUrl=await uploadToImgBB(_pendingFile);
    const note=(document.getElementById('proof-note')||{}).value||'';
    const ts=Date.now();

    let leveledUp=false, newLevel=0;

    S.update(st=>{
      // Apply XP
      st.xp[q.cat]=(st.xp[q.cat]||0)+q.xp;
      st.gold+=q.gold;

      // Recalculate level
      const xp=totalXP(st);
      const newLv=getLevel(xp);
      if(newLv>st.level){ leveledUp=true; newLevel=newLv; st.level=newLv; }

      // Store proof
      st.proofs.push({
        id: `${questId}_${ts}`,
        questId, cat:q.cat,
        xp:q.xp, gold:q.gold,
        imgUrl, note, ts
      });

      // Cooldown
      st.cooldowns[questId]=Date.now()+CFG.QUEST_COOLDOWN_MS;

      // Streak
      const today=todayStr();
      if(st.streak.lastProofDate!==today){
        const yesterday=new Date(Date.now()-86400000).toISOString().slice(0,10);
        if(st.streak.lastProofDate===yesterday) st.streak.count++;
        else st.streak.count=1;
        st.streak.lastProofDate=today;
      }
    });

    // Sync leaderboard
    const s=S.get();
    LB.upsert(s.user.email,{
      name:s.user.name,
      xp:totalXP(s),
      level:getLevel(totalXP(s)),
      proofCount:s.proofs.length
    });

    document.getElementById('modal-bg').style.display='none';
    flashScreen('rgba(184,245,58,0.12)');
    floatXP(q.xp);
    confetti({particleCount:100,spread:70,origin:{y:0.6},colors:['#B8F53A','#60A5FA','#F97316']});
    Toast.show({title:`Proof verified! +${q.xp} XP · +${q.gold} Gold`,type:'success'});

    if(leveledUp){
      setTimeout(()=>showLevelUpModal(newLevel),600);
    }

  }catch(err){
    btn.innerHTML='<i class="fa fa-upload"></i> Submit';
    btn.disabled=false;
    errEl.style.display='block';
    errEl.textContent='Upload failed: '+err.message+'. Check your ImgBB key or try again.';
  }
}
